    public void updateJob(String triggerGroupName, String jobName, String cron) throws SchedulerException{
        //è·å–Scheduler
        Scheduler scheduler = getScheduled();
 
        TriggerKey triggerKey = TriggerKey.triggerKey(jobName, triggerGroupName);
